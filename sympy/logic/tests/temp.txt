(declare-const d1 Bool)
(declare-const d2 Bool)
(declare-const d3 Bool)
(declare-const d4 Bool)
(declare-const z Real)
(declare-const y Real)
(declare-const x Real)
(assert (or d1))
(assert (or d2))
(assert (or d3))
(assert (or d4))
(assert (implies d1 (> (+ x (* -1 y)) 10)))
(assert (implies d2 (> z 5)))
(assert (implies d3 (< (+ x y z) 50)))
(assert (implies d4 (< (+ x (* -1 z)) 30)))
